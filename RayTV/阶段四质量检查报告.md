```typescript
/**
 * 从URL获取配置并解析
 */
public async getParsedConfig(url: string): Promise<ParsedConfig | null> {
  try {
    const config = await this.fetchConfig(url);
    if (!config) {
      return null;
    }
    return await this.parseConfig(config);
  } catch (error) {
    this.logger.error(`Failed to get parsed config from ${url}:`, error);
    return null;
  }
}

/**
 * 从URL获取直播数据
 */
public async getLiveDataFromUrl(url: string): Promise<LiveChannel[]> {
  try {
    const config = await this.fetchConfig(url);
    if (!config || !config.live) {
      return [];
    }
    return config.live;
  } catch (error) {
    this.logger.error(`Failed to get live data from ${url}:`, error);
    return [];
  }
}

// 添加必要的类型定义
export interface ParsedConfig {
  vod?: VodSite[];
  live?: LiveChannel[];
  wallpapers?: Wallpaper[];
  adBlockRules?: AdBlockRule[];
  decoderConfigs?: DecoderConfig[];
}
```

### 🟡 问题8: 错误处理不足
**位置**: 多处

**问题描述**:

| 位置 | 问题 | 影响 |
|------|------|------|
| NetworkSourceManager | 网络错误无重试 | 临时网络问题导致失败 |
| SourcePoolManager | 解析错误无降级 | 单个源错误导致整体失败 |
| NetworkSourceManager | 存储错误静默失败 | 用户数据丢失 |
| ContentAccessService | 搜索失败无提示 | 用户体验差 |

**修复方案**:

1. 添加重试机制:
```typescript
public async loadFromUrl(url: string, retryCount: number = 3): Promise<ConfigSource | null> {
  for (let i = 0; i < retryCount; i++) {
    try {
      return await this.fetchConfig(url);
    } catch (error) {
      if (i === retryCount - 1) {
        throw error;
      }
      await this.sleep(1000 * Math.pow(2, i)); // 指数退避
    }
  }
  return null;
}

private sleep(ms: number): Promise<void> {
  return new Promise(resolve => setTimeout(resolve, ms));
}
```

2. 添加降级逻辑:
```typescript
public async loadFromSource(source: NetworkSource): Promise<SourceContent> {
  try {
    return await this.loadFromUrl(source.url);
  } catch (error) {
    this.logger.warn(`Source ${source.id} failed, using cached data`);
    return this.getCachedData(source.id) || this.getEmptyContent();
  }
}
```

3. 添加存储错误处理:
```typescript
public async saveSources(sources: NetworkSource[]): Promise<void> {
  try {
    await this.storage.set('network_sources', sources);
  } catch (error) {
    this.logger.error('Failed to save sources:', error);
    throw new Error('保存网络源失败，请检查存储权限');
  }
}
```

## 五、架构一致性分析

### 1. 服务依赖关系 ⚠️

**当前依赖图**:
```
NetworkSourceManager → SourcePoolManager
                    ↘
                     → ContentAccessService
                          ↓
                    SourcePoolManager (循环依赖风险!)
```

**问题**:
- ContentAccessService 依赖 SourcePoolManager
- NetworkSourceManager 也依赖 SourcePoolManager
- SourcePoolManager 又依赖 ConfigSourceService（应该）

**建议重构**:
```
NetworkSourceManager → EventBus (发布事件)
ContentAccessService → EventBus (订阅事件)
      ↓
SourcePoolManager (单例，被其他服务调用)
      ↓
ConfigSourceService
```
